\newenvironment{questions}
    {
        \begin{enumerate}[label=\textbf{\thechapter.\arabic*}]
    }
    { 
        \end{enumerate}
    }

\newenvironment{parts}
    {
        \begin{enumerate}[label=(\alph*)]
        \let\part\mipart
    }
    { 
        \end{enumerate}
    }

\newenvironment{multiplechoice}
    {   
        \begin{samepage}
        \nopagebreak % no breaking between choices
        \begin{enumerate}[label=\Alph*.]
    }
    {
        \end{enumerate}
        \end{samepage}
    }
